class Guardian(
    val Nombre:String,
    val Clase:String,
    val Raza:String,

){
    var Poderes:MutableList<String> = mutableListOf()
    fun agregarPoder(poder:String){
        Poderes.add(poder)
        println("Ahora ${Nombre} tiene el poder del/la ${poder} en sus manos\n")
    }
    fun imprimir(){
        println("Guardian")
        println("Nombre: ${Nombre}")
        println("Clase: ${Clase}")
        println("Raza: ${Raza}")
    }
}

    fun imprimir(){

}
